class {class_name}

	def initialize (n_classes, vectors, coeffs, inters, weights, kernel, gamma, coef0, degree)
		@n_classes = n_classes
		@classes = Array.new(n_classes)
		for i in 0 ... n_classes
			@classes[i] = i
		end

		@vectors = vectors
		@coeffs = coeffs
		@inters = inters
		@weights = weights

		@kernel = kernel.upcase
		@gamma = gamma
		@coef0 = coef0
		@degree = degree
	end

	{method}

end
